16. 替换空格

请实现一个函数，把字符串中的每个空格替换成”%20”。

class Solution {
public:
    string replaceSpaces(string &str) {
        string str1;
        for(int i=0;i<str.length();i++){
            if(str[i]==' '){
                str1+="%20";
            }else str1+=str[i];
        }
        return str1;
    }
};

1204. 错误票据

关键点：定行不定量输入

我的方法：用字符串getline(cin,str);substr截取；atoi(str.c_str())转为数字；

曲折：i<s.length()/s.size()报错，推测原因为，两东西类型不一致，添加int len=s.length()解决。
丢失数字，原因最后一行数字最后没有空格，无法判断，添加判断即可，注意最后下标为len-1

可改进部分：输入？存储方式？

#include <bits/stdc++.h>
using namespace std;
int a[100010];
int main(){
    int N,mina=100010,maxa=0;
    cin>>N;
    for(int ii=0;ii<=N;ii++){
        string s;
        getline(cin,s);
        int len=s.size();
        int flagi=0;
        for(int i=0;i<len;i++){
            if(s[i]==' '){
                string str=s.substr(flagi,i-flagi); //截取
                flagi=i+1;  //准备下次
                int aa=atoi(str.c_str());//转换
                a[aa]++;//存储
                maxa=max(aa,maxa);
                mina=min(aa,mina);
            }else if(i==len-1){
                string str=s.substr(flagi,i-flagi+1);
                int aa=atoi(str.c_str());
                a[aa]++;
                maxa=max(aa,maxa);
                mina=min(aa,mina);
            }
        }  
    }
    int x,y;
    for(int i=mina;i<=maxa;i++){
            if(a[i]==0)x=i;
           else if(a[i]==2)y=i;
    }
    cout<<x<<" "<<y;
    return 0;
}

钻空子的读入方式 by.chaosliang :
无视N,直接while(cin >> tp)//直接读到文件尾部停止
while(cin >> tp)是这样理解的：当读到文件结尾时cin>>tp返回值是0，还没到结尾的话返回值是1。这样的话，在oj系统上每一组测试数据其实是一组文件，当读入这组文件到结尾就会跳出循环输出结果。但是在本地的运行的话，应该是在黑屏终端下面的，你输入了很多数据，但是他无法判断你是否输入结束（就是到文件结尾），从而不能跳出循环。这个时候你需要在自己输完之后手动加一个命令ctrl+c或者ctrl+d （平台不同的话两个都试试看），表示到达文件结尾，然后就有答案了。

1208. 翻硬币

比较简单，顺序结构

#include <bits/stdc++.h>
using namespace std;
int main(){
    
    string s1,s2;
    int s=0;
    getline(cin,s1);
    getline(cin,s2);
    int len=s1.size();
    int i=0;
    for(;i<len-1;i++){
        if(s1[i]!=s2[i]){
            s1[i]=s2[i];
            if(s1[i+1]=='*'){s1[i+1]='o';}
            else if(s1[i+1]=='o'){s1[i+1]='*';}
            
            s++;
        }
        
    }

    cout<<s;
}

1210. 连号区间数

错误答案

#include <bits/stdc++.h>
using namespace std;
int a[10001];
int main(){
    int n,s=0;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    for(int i=0;i<n;i++){
        int maxi=a[i],mini=a[i];
        int x[10001]={0};
        for(int j=i;j<n;j++){
            
            x[a[j]]=1;
            if(a[j]>maxi)maxi=a[j];
            if(a[j]<mini)mini=a[j];
            
            for(int z=mini;z<maxi;z++){
                if(x[z]==0)goto a;
                
            }
            cout<<i+1<<" "<<j+1<<endl;
            s++;
        }
a:      continue;
    }
    cout<<s;
    
    
    return 0;
}

可简化：

for(int i=0;i<n;i++){
    int maxi=a[i],mini=a[i];
    
    for(int j=i;j<n;j++){
        
        if(a[j]>maxi)maxi=a[j];
        if(a[j]<mini)mini=a[j];
        if(maxi-mini==j-i)s++;
        else break;
    }

}
cout<<s;

#include<bits/stdc++.h>
using namespace std;
int n;
const int N=11000;
int a[N],ans=0;
int max1=0,min1=11000;
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    {  
        cin>>a[i];
    }
    for(int i=1;i<=n;i++)
    {
        max1=0;
        min1=11000;
        for(int j=i;j<=n;j++)
        {
           max1=max(a[j],max1);
           min1=min(a[j],min1);
           if((max1-min1)==(j-i))
           {
             ans++;
           }
        }
    }
    cout<<ans<<endl;
}
