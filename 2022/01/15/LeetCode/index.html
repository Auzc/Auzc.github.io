13. 罗马数字转整数

暴力：

class Solution {
public:
    int romanToInt(string s) {
        int sum=0;
        for(int i=s.length()-1;i>=0;i--){
            if(s[i]=='I'){
                if((i+1>s.length()||s[i+1]=='V'||s[i+1]=='X')&&(i-1<0||s[i-1]!='I')){
                    sum-=1;
                }else{
                    sum+=1;
                }
                
            }else if(s[i]=='V'){
                sum+=5;
            }else if(s[i]=='X'){
                if((i+1>s.length()||s[i+1]=='L'||s[i+1]=='C')&&(i-1<0||s[i-1]!='X')){
                    sum-=10;
                }else{
                    sum+=10;
                }
            }else if(s[i]=='L'){
                sum+=50;
            }else if(s[i]=='C'){
                if((i+1>s.length()||s[i+1]=='D'||s[i+1]=='M')&&(i-1<0||s[i-1]!='C')){
                    sum-=100;
                }else{
                    sum+=100;
                }
            }else if(s[i]=='D'){
                sum+=500;
            }else if(s[i]=='M'){
                sum+=1000;
            }
        }
        return sum;

    }
};

class Solution {
    public String longestCommonPrefix(String[] strs) {
        if(strs.length==0)return "";
        //公共前缀比所有字符串都短，随便选一个先
        String s=strs[0];
        for (String string : strs) {
            while(!string.startsWith(s)){
                if(s.length()==0)return "";
                //公共前缀不匹配就让它变短！
                s=s.substring(0,s.length()-1);
            }
        }
        return s;
    }
}

class Solution {

        stack<char> mystack;

26. 删除有序数组中的重复项

给你一个有序数组 nums ，请你 原地 删除重复出现的元素，使每个元素 只出现一次 ，返回删除后数组的新长度。
不要使用额外的数组空间，你必须在 原地 修改输入数组 并在使用 O(1) 额外空间的条件下完成。

妙不可言啊

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if (nums.size() < 2) return nums.size();
	    int j = 0;
	    for (int i = 1; i < nums.size(); i++)
		    if (nums[j] < nums[i]) nums[++j] = nums[i];   //j在追赶i，相同时j不变i加。
	    return ++j;

    }
};

27. 移除元素

给你一个数组 nums 和一个值 val，你需要 原地 移除所有数值等于 val 的元素，并返回移除后数组的新长度。
不要使用额外的数组空间，你必须仅使用 O(1) 额外空间并 原地 修改输入数组。
元素的顺序可以改变。你不需要考虑数组中超出新长度后面的元素。

class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        
	    int j = 0;
	    for (int i = 0; i < nums.size(); i++)
		    if (nums[i]!=val) nums[j++] = nums[i];  
	    return j;

    }
};

28. 实现 strStr()

实现 strStr() 函数。
给你两个字符串 haystack 和 needle ，请你在 haystack 字符串中找出 needle 字符串出现的第一个位置（下标从 0 开始）。如果不存在，则返回 -1 。

暴力

class Solution {
public:
    int strStr(string haystack, string needle) {
        if(haystack.length()==0&&needle.length()==0)return 0;
        int len=needle.length(),i;
        if(haystack.length()<needle.length())return -1;
        
        for(i=0;i<haystack.length();i++){
            
            if(haystack.substr(i,len)==needle){
                return i;
            }if(i>=haystack.length()-len)break;

        }
        return -1;

    }
};

35. 搜索插入位置

给定一个排序数组和一个目标值，在数组中找到目标值，并返回其索引。如果目标值不存在于数组中，返回它将会被按顺序插入的位置。
请必须使用时间复杂度为 O(log n) 的算法。
暴力：

class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        for(int i=0;i<nums.size();i++){
            if(nums[i]>=target){
                return i;
            }
        }
        return nums.size();

    }
};

二分法

class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n = nums.size();
        int left = 0;
        int right = n - 1; // 定义target在左闭右闭的区间里，[left, right]
        while (left <= right) { // 当left==right，区间[left, right]依然有效
            int middle = left + ((right - left) / 2);// 防止溢出 等同于(left + right)/2
            if (nums[middle] > target) {
                right = middle - 1; // target 在左区间，所以[left, middle - 1]
            } else if (nums[middle] < target) {
                left = middle + 1; // target 在右区间，所以[middle + 1, right]
            } else { // nums[middle] == target
                return middle;
            }
        }
        // 分别处理如下四种情况
        // 目标值在数组所有元素之前  [0, -1]
        // 目标值等于数组中某一个元素  return middle;
        // 目标值插入数组中的位置 [left, right]，return  right + 1
        // 目标值在数组所有元素之后的情况 [left, right]， return right + 1
        return right + 1;
    }
};

53. 最大子数组和

给你一个整数数组 nums ，请你找出一个具有最大和的连续子数组（子数组最少包含一个元素），返回其最大和。
子数组 是数组中的一个连续部分。

暴力

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        long long int sum=nums[0],max=sum,temp_i=0;
        for(int i=0;i<nums.size();i++){
            if(nums[i]>=0){
                sum=nums[i];if(sum>max){max=sum;}
                temp_i=i;
                while(sum>0){
                    temp_i++;
                    if(temp_i>=nums.size())break;
                    sum+=nums[temp_i];
                    if(sum>max){max=sum;}
                    if(temp_i==nums.size()-1) goto end;   //不加这句，就超时了，后面都是无效比较  测试例子中10000个10000的情况就无语。
                }
            }else if(nums[i]>max){max=nums[i];}

        }
        end:
        return max;

    }
};

大佬简化

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int res = nums[0];
        int sum = 0;
        for (int num : nums) {
            if (sum > 0)
                sum += num;
            else
                sum = num;      //如果前一位数是负数，就会在这里判断，取这个正数，妙啊。
            if(sum>res){res=sum;} //存储了峰值
        }
        return res;

    }
};

该算法更为简便之处是忽略了对子序列的寻找比较,而是根据规律直接找出最佳答案.
对于含有正数的序列而言,最大子序列肯定是正数,所以头尾肯定都是正数.我们可以从第一个正数开始算起,每往后加一个数便更新一次和的最大值;当当前和成为负数时,则表明此前序列无法为后面提供最大子序列和,因此必须重新确定序列首项.

66. 加一

给定一个由 整数 组成的 非空 数组所表示的非负整数，在该数的基础上加一。
最高位数字存放在数组的首位， 数组中每个元素只存储单个数字。
你可以假设除了整数 0 之外，这个整数不会以零开头。

百家之长：

class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        for(int i=digits.size()-1;i>=0;i--){
            if(digits[i]!=9){
                digits[i]++;
                return digits;
            }
            digits[i]=0;
        }
        //所有为9的情况
        digits.insert(digits.begin() , 1);
        return digits;
    }
};

67. 二进制求和

给你两个二进制字符串，返回它们的和（用二进制表示）。
输入为 非空 字符串且只包含数字 1 和 0。

补齐0；预留一位首0；删去首0。

class Solution {
public:
    string addBinary(string a, string b) {
        if(a.length()>b.length()) swap(a,b);
        int len_ab=b.length()-a.length();

        char cc=b.length()-a.length()+'0';
        for(int i=0;i<len_ab+1;i++){
            a="0"+a;
        }

        b="0"+b;
        for(int i=a.length()-1;i>=0;i--){
        
            int sum=(b[i]-'0')+(a[i]-'0');
            switch(sum){
                case 0:b[i]='0';break;
                case 1:b[i]='1';break;
                case 2:b[i]='0';b[i-1]=((b[i-1]-'0')+1)+'0';break;
                case 3:b[i]='1';b[i-1]=((b[i-1]-'0')+1)+'0';break;
                default:break;
            }
        }

        while(b.length()>1){
            if(b[0]=='0'){
                b=b.substr(1);   
            }else break;
        }
        
        return b;

    }
};

69. Sqrt(x)

算术平方根，返回整数

二分法：

class Solution {
public:
    int mySqrt(int x) {
        if(x == 1)
            return 1;
        int min = 0;
        int max = x;
        while(max-min>1)
        {
            int m = (max+min)/2;
            if(x/m<m)           //二分查找，用x/m<m而不是m*m>x防止溢出
                max = m;
            else
                min = m;
        }
        return min;

    }
};

内置函数：

return (int)sqrt(x);

136. 只出现一次的数字

给定一个非空整数数组，除了某个元素只出现一次以外，其余每个元素均出现两次。找出那个只出现了一次的元素。
异或^:
（1）与0相^，保留原值
（2）交换两个值，不用临时变量
（3）与自己相^，值为0

class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int re=0;
        for(int i=0;i<nums.size();i++)
        {
            re=re^nums[i];
        }
        return re;

    }
};
